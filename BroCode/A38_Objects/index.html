// object = A collection of related properties and/or methods
            //                Can represent real world objects (people, products, places)
            //                object = {key:value,
            //                                 function()}

            const person1 = {
                firstName : "Spongebob",
                lastName : "Squarepants",
                age : 32,
                isEmployed : true,
                greet : () => {console.log("Hello im Spongebob")},
                eat: function(){console.log("I am eating a Krabby Patty")},
            }

            const person2 = {
                firstName : "Patrick",
                lastName : "Star",
                age : 42,
                isEmployed : true,
                greet : () => {console.log("Hello im Patrick")},
                eat: function(){console.log("I am eating roast beef, chicken, and pizza")},
            }

            console.log(`Person 1 object => `,person1);
            console.log(`Person 2 object => `,person2);


            console.log(`Function call :- `)
            person1.greet();
            person1.eat();

            person2.greet();
            person2.eat();
