// inheritance = allows a new class to inherit properties and methods 
//                         from an existing class (parent -> child)
//                         helps with code reusability

class Animal{
    alive = false;

    constructor(){
        this.alive = true;
    }

    eat(){
        return `${this.name} the ANIMAL is eating !`
    }

    sleep(){
        return `Sushh ! ${this.name} the ANIMAL is sleeping !`
    }
}

class Rabbit extends Animal{
    
    name = "Bucks Bunny";

    isAlive(){
        return this.alive
    }
    run(){
        return `${this.name} the RABBIT is HOPPING !`
    }

}

class Fish extends Animal{
    
    name = "Nemo";
    
    isAlive(){
        return this.alive
    }
    swim(){
        return `${this.name} the FISH is SWIMMING !`
    }

}

class Hawk extends Animal{
    
    name = "Maui";
      
    isAlive(){
        return this.alive
    }
    fly(){
        return `${this.name} the HAWK is FLYING !`
    }

}

const rabbit = new Rabbit();
const fish = new Fish();
const hawk = new Hawk();
