for each object, we create a common constructor, which goes like :-

function Car(make,model,year,color){
    this.make = make,
    this.model = model,
    this.year = year,
    this.color = color

    this.drive = function(ignition){
        if (ignition === `on` || ignition === `ON`){
            console.log(`You are driving a ${this.year} ${this.make} ${this.model}...!`);
        }else if(ignition === `off` || ignition === `OFF`){
            console.log(`you TURNED OFF ${this.model} !`);
        }else{
            console.log(`TURN ON TO DRIVE A ${this.year} ${this.make} ${this.model} !`);
        }
    }
}

console.log(`Car (constructor =>)`,Car)

const car1 = new Car("Ford","Mustang",1989,"Black")
console.log(`car1 => `,car1)
car1.drive("ON")
